    private function verdadero($elargs){
//        $elargs = func_get_args();
        if($elargs[0] == 0){
            return 0;
        }
        $value=0;
        foreach ($elargs as $answerOfQuestion){
            $value+= $this->results[$answerOfQuestion-1]->answer;
        }
        return $value;
    }

    private function falso($elargs){
//        $elargs = func_get_args();
        if($elargs[0] == 0){
            return 0;
        }
        $value=0;
        foreach ($elargs as $answerOfQuestion){
            $value+= !$this->results[$answerOfQuestion-1]->answer;
        }
        return $value;
    }
